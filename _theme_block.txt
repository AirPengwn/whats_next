<script>
(function(){
  var saved=localStorage.getItem('erin_theme');
  var prefersDark=window.matchMedia&&window.matchMedia('(prefers-color-scheme:dark)').matches;
  if(saved==='dark'||(!saved&&prefersDark)) document.documentElement.classList.add('dark');
  function upBtn(){var b=document.getElementById('theme-toggle');if(b)b.innerHTML=document.documentElement.classList.contains('dark')?'&#9728;':'&#127769;';}
  document.addEventListener('DOMContentLoaded',upBtn);
  window.__toggleTheme=function(){var d=document.documentElement.classList.toggle('dark');localStorage.setItem('erin_theme',d?'dark':'light');upBtn();};
})();
</script>
